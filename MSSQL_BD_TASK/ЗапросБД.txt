CREATE DATABASE AutoSalonDB;

CREATE TABLE Brands (
    BrandID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),    
    BrandName NVARCHAR(100) NOT NULL,                       
    Country NVARCHAR(50) NOT NULL,                           
    IsActive BIT NOT NULL DEFAULT 1,                       
    FoundedYear INT NULL,                                    
    CreatedAt DATETIME2 NOT NULL DEFAULT GETDATE()           
);
GO

CREATE TABLE Models (
    ModelID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),   
    ModelName NVARCHAR(100) NOT NULL,                        
    BrandID UNIQUEIDENTIFIER NOT NULL,                       
    BodyType NVARCHAR(50) NOT NULL,                          
    BasePrice DECIMAL(12,2) NOT NULL,                      
    IsActive BIT NOT NULL DEFAULT 1,                        
    
    CONSTRAINT FK_Models_Brands 
    FOREIGN KEY (BrandID) REFERENCES Brands(BrandID)
);
GO

CREATE TABLE Cars (
    CarID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),     
    VIN NVARCHAR(17) NOT NULL,                              
    ModelID UNIQUEIDENTIFIER NOT NULL,                       
    Color NVARCHAR(30) NOT NULL,                            
    YearOfManufacture INT NOT NULL,                         
    Mileage INT NOT NULL DEFAULT 0,                          
    Price DECIMAL(12,2) NOT NULL,                           
    IsSold BIT NOT NULL DEFAULT 0,                          
    ArrivalDate DATETIME2 NOT NULL DEFAULT GETDATE(),        
    DiscountPercent DECIMAL(4,2) NULL                        
);
GO

CREATE UNIQUE INDEX UX_Cars_VIN 
ON Cars(VIN);
GO

-- клиенты
CREATE TABLE Customers (
    CustomerID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(), 
    FirstName NVARCHAR(50) NOT NULL,                         
    LastName NVARCHAR(50) NOT NULL,                         
    Phone NVARCHAR(20) NOT NULL,                             
    Email NVARCHAR(100) NULL,                                
    IsVIP BIT NOT NULL DEFAULT 0,                            
    RegistrationDate DATETIME2 NOT NULL DEFAULT GETDATE()    
);
GO

CREATE TABLE Sales (
    SaleID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),     
    CarID UNIQUEIDENTIFIER NOT NULL,                        
    CustomerID UNIQUEIDENTIFIER NOT NULL,                
    SaleDate DATETIME2 NOT NULL DEFAULT GETDATE(),          
    FinalPrice DECIMAL(12,2) NOT NULL,                    
    PaymentMethod NVARCHAR(50) NOT NULL,                     
    IsCompleted BIT NOT NULL DEFAULT 0,                      
    
    CONSTRAINT FK_Sales_Cars 
    FOREIGN KEY (CarID) REFERENCES Cars(CarID),
    
    CONSTRAINT FK_Sales_Customers 
    FOREIGN KEY (CustomerID) REFERENCES Customers(CustomerID)
);
GO

--доп услуги
CREATE TABLE Services (
    ServiceID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),  
    ServiceName NVARCHAR(100) NOT NULL,                      
    Price DECIMAL(10,2) NOT NULL,                            
    IsActive BIT NOT NULL DEFAULT 1,                         
    Description NVARCHAR(500) NULL                           
);
GO

-- промежуточная таблицпа для связи продаж и услуг 
CREATE TABLE SalesServices (
    SaleID UNIQUEIDENTIFIER NOT NULL,
    ServiceID UNIQUEIDENTIFIER NOT NULL,
    ServicePrice DECIMAL(10,2) NOT NULL,                    
    
    CONSTRAINT PK_SalesServices PRIMARY KEY (SaleID, ServiceID),
    
    CONSTRAINT FK_SalesServices_Sales 
    FOREIGN KEY (SaleID) REFERENCES Sales(SaleID),
    
    CONSTRAINT FK_SalesServices_Services 
    FOREIGN KEY (ServiceID) REFERENCES Services(ServiceID)
);

SELECT 
    b.BrandName,
    m.ModelName,
    c.Color,
    c.YearOfManufacture,
    c.Price,
    c.DiscountPercent
FROM Cars c
INNER JOIN Models m ON c.ModelID = m.ModelID
INNER JOIN Brands b ON m.BrandID = b.BrandID
WHERE c.IsSold = 0 
  AND m.BodyType = 'Кроссовер' 
  AND c.Price > 3000000
ORDER BY c.Price DESC;

-- Обновление данных
-- Применить скидку 5% ко всем автомобилям старше 2024 года с пробегом более 10000 км
UPDATE Cars
SET DiscountPercent = 5.00
WHERE YearOfManufacture < 2025 
  AND Mileage > 10000 
  AND IsSold = 0;

-- Удаление данных
-- Удалить неактивные услуги (которые больше не предоставляются)
DELETE FROM SalesServices 
WHERE ServiceID IN (
    SELECT ServiceID FROM Services WHERE IsActive = 0
);

DELETE FROM Services 
WHERE IsActive = 0;

-- Выборка с группировкой 
-- Статистика продаж по маркам автомобилей
SELECT 
    b.BrandName,
    COUNT(s.SaleID) AS TotalSales,
    SUM(s.FinalPrice) AS TotalRevenue,
    AVG(s.FinalPrice) AS AvgSalePrice,
    MIN(s.FinalPrice) AS MinSalePrice,
    MAX(s.FinalPrice) AS MaxSalePrice
FROM Sales s
INNER JOIN Cars c ON s.CarID = c.CarID
INNER JOIN Models m ON c.ModelID = m.ModelID
INNER JOIN Brands b ON m.BrandID = b.BrandID
WHERE s.IsCompleted = 1
GROUP BY b.BrandName
HAVING COUNT(s.SaleID) >= 1
ORDER BY TotalRevenue DESC;

-- LEFT JOIN
-- Найти всех клиентов и их покупки
SELECT 
    c.FirstName + ' ' + c.LastName AS CustomerName,
    c.Phone,
    c.IsVIP,
    s.SaleDate,
    b.BrandName + ' ' + m.ModelName AS CarPurchased,
    s.FinalPrice
FROM Customers c
LEFT JOIN Sales s ON c.CustomerID = s.CustomerID
LEFT JOIN Cars ca ON s.CarID = ca.CarID
LEFT JOIN Models m ON ca.ModelID = m.ModelID
LEFT JOIN Brands b ON m.BrandID = b.BrandID
ORDER BY c.LastName, c.FirstName;

-- RIGHT JOIN
-- Найти все автомобили и информацию об их продаже 
SELECT 
    b.BrandName,
    m.ModelName,
    c.VIN,
    c.Color,
    c.Price,
    c.IsSold,
    s.SaleDate,
    cust.FirstName + ' ' + cust.LastName AS BuyerName
FROM Sales s
RIGHT JOIN Cars c ON s.CarID = c.CarID
LEFT JOIN Models m ON c.ModelID = m.ModelID
LEFT JOIN Brands b ON m.BrandID = b.BrandID
LEFT JOIN Customers cust ON s.CustomerID = cust.CustomerID
ORDER BY c.IsSold, b.BrandName;

-- INNER JOIN (пересечение)
-- Найти VIP-клиентов, которые совершили покупки
SELECT DISTINCT
    c.FirstName + ' ' + c.LastName AS VIPCustomer,
    c.Email,
    COUNT(s.SaleID) OVER(PARTITION BY c.CustomerID) AS PurchaseCount,
    SUM(s.FinalPrice) OVER(PARTITION BY c.CustomerID) AS TotalSpent
FROM Customers c
INNER JOIN Sales s ON c.CustomerID = s.CustomerID
WHERE c.IsVIP = 1 AND s.IsCompleted = 1;

--Подсчет дополнительных услуг по каждой продаже
SELECT 
    s.SaleID,
    c.FirstName + ' ' + c.LastName AS CustomerName,
    b.BrandName + ' ' + m.ModelName AS CarModel,
    s.FinalPrice AS CarPrice,
    COUNT(ss.ServiceID) AS ServicesCount,
    SUM(ss.ServicePrice) AS TotalServicesPrice,
    s.FinalPrice + ISNULL(SUM(ss.ServicePrice), 0) AS TotalDealPrice
FROM Sales s
INNER JOIN Customers c ON s.CustomerID = c.CustomerID
INNER JOIN Cars ca ON s.CarID = ca.CarID
INNER JOIN Models m ON ca.ModelID = m.ModelID
INNER JOIN Brands b ON m.BrandID = b.BrandID
LEFT JOIN SalesServices ss ON s.SaleID = ss.SaleID
GROUP BY s.SaleID, c.FirstName, c.LastName, b.BrandName, m.ModelName, s.FinalPrice;

-- Анализ доступных автомобилей по категориям
SELECT 
    m.BodyType,
    b.BrandName,
    COUNT(c.CarID) AS AvailableCount,
    AVG(c.Price) AS AvgPrice,
    MIN(c.Price) AS MinPrice,
    MAX(c.Price) AS MaxPrice
FROM Cars c
INNER JOIN Models m ON c.ModelID = m.ModelID
INNER JOIN Brands b ON m.BrandID = b.BrandID
WHERE c.IsSold = 0 AND m.IsActive = 1
GROUP BY m.BodyType, b.BrandName
ORDER BY m.BodyType, AvgPrice DESC;
